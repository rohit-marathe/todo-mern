```bash
# Clone this repository
$ git clone https://github.com/rohit-marathe/todo-mern.git

# Change the directory to backend
$ cd backend

# Then Install dependencies
$ npm install

# Change the directory to frontend
$ cd frontend

# Then Install dependencies
$ npm install

# To start the backend server change the directory to backend
$ cd backend

# Then start the server
$ npm start

# To start the frontend server change the directory to frontend
$ cd frontend

# Then start the server
$ npm start

```
